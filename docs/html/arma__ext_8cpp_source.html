53  int size = B.n_cols;
54  int number = d.size();
55  int diag_size;

57  colvec column;
58  mat output;

62  QVector<int> subdiagonals;

64

70 
71 
72  for (i = 0; i < number; ++i){

84

86  for (i = 0; i < diag_size; ++i){
87  output.diag(k)(i) = column(i);
88  }
89  }
90 
91  else {
92  for (i = 0; i < column_size; ++i){
93  output.diag(k)(i) = column(i);
94  }
95  }
96  }
97 
98  for (i = 0; i < superdiagonals.size(); ++i){
99  k = superdiagonals[i];

101  column = B.col(i);
102  if (diag_size < column_size){
103  j = column_size;

105  output.diag(k)(i) = column(j);
106  --j;
107  }
108  }

112  output.diag(k)(i) = column(i);

175 bool arma_ext::svds(mat X, int k, mat &U, vec &s, mat &V)
176 {
177  if (X.n_cols < 2){
178  cerr << "svds: X must be 2D matrix" << endl;
179  //throw SVDS_Dimension_X;
180  return false;
181  }
182 
183  double root_2 = sqrt(2.0);
184  double tolerance = 1e-10 / root_2; //tolerance for convergence (ARPACK default)
185  double epsilon = datum::eps; //eps in Octave and MATLAB
186  int m = X.n_rows;
187  int n = X.n_cols;
188  int p = std::min(m, n); //used below to establish tolerances.
189  int q = std::max(m, n);
190 
191  if (k > p){
192  if (k > m)
193  cerr << "svds: value of k " << k << "is greater than number of rows " << m << endl;
194  else
195  cerr << "svds: value of k" << k << "is greater than number of columns " << n << endl;
196  }
197 
198  k = std::min(p, k); //make sure value of k is less than smallest dimension
199 
200  sp_mat B(m+n, m+n);
201  B.submat(span(0, m-1), span(m, m+n-1)) = X; // top right corner
202  B.submat(span(m, m+n-1), span(0, m-1)) = X.t(); //bottom left corner
203 
204  //mat B_full(B.n_rows, B.n_cols);
205  //for(int i = 0; i < B.n_rows; ++i)
206  // for (int j = 0; j < B.n_cols; ++j)
207  // B_full(i,j) = B(i, j); //used to test copying of matrix
208  //B_full.save("B.csv", csv_ascii);
209 
210  //If, for some reason, a matrix of zeroes is input...
211  if (B.n_nonzero == 0){
212  U = eye(m, k);
213  s = zeros(k, k);
214  V = eye(n, k);
215  cerr << "svds: input matrix has no non-zero elements" << endl;
216  //throw SVDS_Zeros;
217  return false;
218  }
219 
220  vec eigval;
221  mat eigvec;
222 
223  //because we're using sigma=0, results of eigs will be centered around 0
224  //we throw out the negative ones, then add in 0 eigenvalues if we have too
225  //few
226  bool eigs_success = eigs_sym(eigval, eigvec, B, k*2);
227  //cout << "eigval=" << eigval << endl;
228  //cout << "eigvec=" << eigvec << endl;
229 
230  double eigval_max = eigval.max();
231 
232 
233 
234  if (!eigs_success){
235  cerr << "svds: arma::eigs_sym did not converge!" << endl;
236  }
237 
238  //Manipulate the results
239 
240  //vector of indices of the eigenvalues, sorted in descending order
241  uvec sorted_eigval_indices = sort_index(eigval, "descend");
242 
243  //sort eigvec from largest to smallest
244  eigvec = eigvec.cols(sorted_eigval_indices);
245 
246  double d_tolerance, uv_tolerance; //tolerances for D and U, V
247  if (eigs_success){
248  d_tolerance = q*n*sqrt(epsilon);
249  uv_tolerance = m*sqrt(sqrt(epsilon));
250  }
251  else{
252  d_tolerance = q*eigval_max*epsilon;
253  uv_tolerance = m*sqrt(epsilon);
254 
255  }
256 
257  //cout << "UU" << endl;
258  mat UU = trans(eigvec.rows(0, m-1)) * eigvec.rows(0, m-1);
259  vec diag_UU = UU.diag();
260  //cout << "VV" << endl;
261  mat VV = trans(eigvec.rows(m, m+n-1)) * eigvec.rows(m, m+n-1);
262  vec diag_VV = VV.diag();
263 
264 
265  //find all indices that satisfy tolerances
266  //See the Octave documentation for svds:
267  // We wish to exclude all eigenvalues that are less than zero as these
268  // are artifacts of the way the matrix passed to eigs is formed. There
269  // is also the possibility that the value of sigma chosen is exactly
270  // a singular value, and in that case we're dead!! So have to rely on
271  // the warning from eigs. We exclude the singular values which are
272  // less than or equal to zero to within some tolerance scaled by the
273  // norm since if we don't we might end up with too many singular
274  // values.
275 
276  //cout << "call to find()" << endl;
277  uvec query = find((eigval > d_tolerance) && (abs(diag_UU) > uv_tolerance) && (abs(diag_VV) > uv_tolerance));
278  //cout << query << endl;
279 
280 
281  int number_of_indices = query.n_elem;
282  int end = std::min(k, number_of_indices) - 1;
283  //cout << "end=" << end;
284  //cout << "query.n_elem=" << query.n_elem << endl;
285 
286 
287  uvec return_indices = query.subvec(0, end);
288  //cout << "return_indices " << return_indices << endl;
289  U = sqrt(2) * (eigvec.rows(0, m-1));
290  U = U.cols(return_indices);
291 
292  s = eigval.rows(return_indices);
293 
294  V = sqrt(2) * eigvec.rows(m, m+n-1);
295  V = V.cols(return_indices);
296 
297  //B may have some eigenvalues that are 0 that needed to be included if the
298  //number of non_zero eigenvalues is too small. Will only add eigenvalues in if
299  //they exist (obviously).
300  if (return_indices.n_elem < k){
301  uvec zero_indices = find(abs(eigval) <= d_tolerance);
302  if (zero_indices.n_elem != 0){
303  mat eigvec_sub1 = eigvec.rows(0, m-1);
304  mat eigvec_sub2 = eigvec.rows(m, m+n-1);
305  mat QWU = orth(eigvec_sub1.cols(zero_indices));
306  mat QWV = orth(eigvec_sub2.cols(zero_indices));
307  int n_zero = std::min(QWV.n_cols, std::min(QWU.n_cols, (return_indices.n_elem)));
308 
309  //necessary number of zero eigenvalues
310  vec zero_eigvals = abs(eigval.cols(zero_indices.rows(0, n_zero)));
311 
312  U.insert_cols(U.n_cols, QWU.cols(0, n_zero));
313  s.insert_rows(s.n_rows, zero_eigvals);
314  V.insert_cols(V.n_cols, QWV.cols(0, n_zero));
315  }
316  }
317 
318 
319  uvec indices = sort_index(s, "descend");
320  s = sort(s, "descend");
321  //cout << "s = " << endl << s << endl;
322  //cout << "diagmat(s) = " << endl << diagmat(s) << endl;
323 
324  U = U.cols(indices);
325  V = V.cols(indices);
326  //test for convergence
327 
328  return eigs_success;
329 }

351 bool arma_ext::plsregress(mat X, mat Y, int components,
352  mat &X_loadings, mat &Y_loadings,
353  mat &X_scores, mat &Y_scores,
354  mat &coefficients, mat &percent_variance, mat &fitted)
355 {
356  uword observations = X.n_rows;
357  uword predictors = X.n_cols;
358  uword responses = Y.n_cols;
359  //Mean centering
360  mat Xmeans = arma::mean(X);
361  mat Ymeans = arma::mean(Y);
362  //Octave does below with bsxfun. After optimization, this should hopefully not
363  // be slower.
364  X.each_row() -= Xmeans; //element-wise subtraction of mean
365  Y.each_row() -= Ymeans; //same
366 
367  mat S = trans(X) * Y;
368  mat R = zeros(predictors, components);
369  mat P = R;
370  mat V = R;
371  mat T = zeros(observations, components);
372  mat U = T;
373  mat Q = zeros(responses, components);
374  mat eigvec;
375  vec eigval;
376 
377  mat q;
378  mat r;
379  mat t;
380  mat nt;
381  mat p;
382  mat u;
383  mat v;
384  double max_eigval;
385  for (int i = 0; i < components; ++i){
386  eig_sym(eigval, eigvec, (trans(S) * S));
387  max_eigval = eigval.max();
388  uvec dom_index = find(eigval >= max_eigval);
389  uword dominant_index = dom_index(0);
390 
391  q = eigvec.col(dominant_index);
392 
393  r = S*q; //X block factor weights
394  t = X*r; //X block factor weights
395  t.each_row() -= mean(t); //center t
396  nt = arma::sqrt(trans(t)*t); //compute norm (is arma::norm() the same?)
397  t.each_row() /= nt;
398  r.each_row() /= nt; //normalize
399 
400  p = trans(X)*t; //X block factor loadings
401  q = trans(Y)*t; //Y block factor loadings
402  u = Y*q; //Y block factor scores
403  v = p;
404 
405  //Ensure orthogonality
406  if (i > 0){
407  v = v - V*(trans(V)*p);
408  u = u - T*(trans(T)*u);
409  }
410  v.each_row() /= arma::sqrt(trans(v) * v); //normalize orthogonal loadings
411  S = S - v * (trans(v)*S); //deflate S wrt loadings
412  R.col(i) = r;
413  T.col(i) = t;
414  P.col(i) = p;
415  Q.col(i) = q;
416  U.col(i) = u;
417  V.col(i) = v;
418  }
419 
420  //Regression coefficients
421  mat B = R*trans(Q);
422  fitted = T*trans(Q);
423  fitted.each_row() += Ymeans;
424 
425  //Octave creates copies from inputs before sending to output. Doing same
426  //here just to be safe.
427  coefficients = B;
428  X_scores = T;
429  X_loadings = P;
430  Y_scores = U;
431  Y_loadings = Q;
432  //projection = R;
433  percent_variance.set_size(2, coefficients.n_cols);
434  percent_variance.row(0) = sum(arma::abs(P)%arma::abs(P)) / sum(sum(arma::abs(X)%arma::abs(X)));
435  percent_variance.row(1) = sum(arma::abs(Q)%arma::abs(Q)) / sum(sum(arma::abs(Y)%arma::abs(Y)));
436  cout << "end of plsregress" << endl;
437  return true;
438 
439 }

453 bool arma_ext::VCA(mat X, int endmembers, uvec &indices,
454  mat &endmember_spectra, mat &projected_data, mat &fractional_abundances)
455 {
456  indices.set_size(endmembers);
457  indices.zeros();
458  int m = X.n_rows;
459 
460  mat U;
461  vec s;
462  mat V;
463  svds(trans(X) * X / m, endmembers, U, s, V);
464  mat x_p = trans(U) * trans(X);
465  projected_data = U * x_p;
466  mat x_p_mean = mean(x_p, 0);
467  mat t1 = sum(x_p * trans(x_p_mean));
468  mat y_p = x_p / t1(0);
469 
470 
471  mat A = zeros(endmembers, endmembers);
472  A(endmembers-1, 0) = 1;
473  mat w;
474  mat f;
475  mat v;
476  uvec query;
477  for (int i = 0; i < endmembers; ++i){
478  w = randu<mat>(endmembers, 1);
479  f = w - A * pinv(A) * w;
480  mat sum_squares = sqrt(sum(square(f)));
481  f /= sum_squares(0);
482  v = abs(trans(f) * y_p);
483  query = find(v == v.max());
484  indices(i) = query(0);
485  A.col(i) = y_p.col(indices(i));
486  }
487  endmember_spectra = projected_data.cols(indices);
488  fractional_abundances = trans((pinv(endmember_spectra) * projected_data));
489  //put endmember spectra in the same format used by rest of program.
490  return true;
