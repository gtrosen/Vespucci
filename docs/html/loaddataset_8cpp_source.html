61 void LoadDataset::on_buttonBox_accepted()
62 {
63 
64  QCheckBox *swap_check_box = this->findChild<QCheckBox *>("swapCheckBox");
65  QLineEdit *filename_line_edit = this->findChild<QLineEdit *>("filenameBox");
66  QLineEdit *name_box = this->findChild<QLineEdit *>("nameBox");
67  QLineEdit *y_description_box = this->findChild<QLineEdit *>("yDescription");
68  QLineEdit *x_description_box = this->findChild<QLineEdit *>("xDescription");
69  QComboBox *y_units_box = this->findChild<QComboBox *>("yUnits");
70  QComboBox *x_units_box = this->findChild<QComboBox *>("xUnits");
71 
72  QString y_description = y_description_box->text() + " (" + y_units_box->currentText() + ")";
73  QString x_description = x_description_box->text() + " (" + x_units_box->currentText() + ")";
74 
75  QString name = name_box->text();
76  QString filename = filename_line_edit->text();
77  QFileInfo file_info(filename);
78 
79  //bool remove_at_position = false;
80  //int temp_position=0;
81  //int position;
82  bool swap = swap_check_box->isChecked();
83 
84  QStringList names_list = workspace->dataset_names();
85  QStringList::iterator i;
86 
87  int warning_response=QMessageBox::Yes;
88  //Warning message pops up when user enters a name that already exists
89  for (i=names_list.begin(); i != names_list.end(); ++i){
90  if (name==(*i)){
91  warning_response =
92  QMessageBox::question(this,
93  "Duplicate Name",
94  "There is already a dataset with this"
95  " name in the workspace. Datasets are"
96  " not indexed by name, but using two"
97  " datasets with the same name may lead"
98  " to confusion. Are you sure you wish"
99  " to continue with this name?",
100  QMessageBox::No,
101  QMessageBox::Yes);
102  }
103  }
104 
105  if (warning_response == QMessageBox::Yes && file_info.exists()){
106  if (file_info.suffix() == "txt"){
107  QFile inputfile(filename);
108  inputfile.open(QIODevice::ReadOnly);
109  QTextStream inputstream(&inputfile);
110  QSharedPointer<SpecMap> data(new SpecMap(inputstream,
111  workspace->main_window(),
112  directory_,
113  swap));
114  inputfile.close();
115  if (!data->ConstructorCancelled()){
116  data->SetName(name);
117  data->SetXDescription(x_description);
118  data->SetYDescription(y_description);
119  workspace->AddDataset(data);
120  workspace->set_directory(file_info.dir().absolutePath());
121  }
122  }
123 
124  if (file_info.suffix() == "vds"){
125  QMessageBox::critical(this, "Feature not Implemented", "This file type is not supported yet.");
126  return;

134  //}
135  }

137 
138  }
139 }
