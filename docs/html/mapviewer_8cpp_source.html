167 void MapViewer::on_actionSave_Image_As_triggered()
168 {
169  QString path = *directory_;
170  path = path + "/" + name_;
171 
172  QString filename =
173  QFileDialog::getSaveFileName(this,
174  tr("Save File"),
175  path,
176  tr("Tagged Image File Format (*.tif);; "
177  "Windows Bitmap (*.bmp);; "
178  "Portable Network Graphics (*.png);; "
179  "JPEG (*.jpg)"));
180 
181  QStringList filename_list = filename.split(".");
182  QString extension = filename_list.last();
183 
184  //this method of determining type may not be valid on non-Win platforms
185  //check this on GNU/Linux and Mac OSX later.
186 
187  if (extension == "bmp")
188  parent_->saveBmp(filename, 0, 0, 1.0);
189 
190  else if (extension == "png"){
191  bool ok;
192  int quality = QInputDialog::getInt(this, "Enter Quality",
193  "Quality (%)",
194  80, 0, 100, 1, &ok);
195  if (ok)
196  parent_->savePng(filename, 0, 0, 1.0, quality);
197  }
198 
199  else if (extension == "jpg"){
200  bool ok;
201  int quality = QInputDialog::getInt(this, "Enter Quality",
202  "Quality (%)",
203  80, 0, 100, 1, &ok);
204  if (ok)
205  parent_->saveJpg(filename, 0, 0, 1.0, quality);
206  }
207 
208  else{
209  //default to tif, force extension (for Windows compatability)
210  if (extension != "tif")
211  filename.append(".tif");
212  bool ok;
213  int quality = QInputDialog::getInt(this,
214  "Compression",
215  "Enter 0 for no compression,"
216  "1 for LZW lossless compression",
217  0, 0, 1, 1, &ok);
218  if (ok)
219  parent_->saveTiff(filename, 0, 0, 1.0, quality);
220  }
221 
222 }
