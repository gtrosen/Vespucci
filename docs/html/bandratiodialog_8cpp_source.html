74 void BandRatioDialog::on_buttonBox_accepted()
75 {
76  double first_entered_min = first_min_box_->text().toDouble();
77  double second_entered_min = second_min_box_->text().toDouble();
78  double first_entered_max = first_max_box_->text().toDouble();
79  double second_entered_max = second_max_box_->text().toDouble();
80 
81  vector<int> first_range =
82  data_->FindRange(first_entered_min, first_entered_max);
83  vector<int> second_range =
84  data_->FindRange(second_entered_min, second_entered_max);
85 
86  int first_min = first_range[0];
87  int first_max = first_range[1];
88  int second_min = second_range[0];
89  int second_max = second_range[1];
90 
91  QString name = name_box_->text();
92  QString value_method = value_method_selector_->currentText();
93  QString integration_method = integration_method_selector_->currentText();
94  int color_index = color_selector_->currentIndex();
95 
96  data_->BandRatio(first_min,
97  first_max,
98  second_min,
99  second_max,
100  name,
101  value_method,
102  integration_method,
103  color_index);
104 }
