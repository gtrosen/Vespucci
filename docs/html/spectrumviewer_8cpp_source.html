159 void SpectrumViewer::MapClicked(QCPAbstractPlottable *plottable, QMouseEvent *event)
160 {
161  unsigned int i;
162  double x = (double) event->x();
163  double y = (double) widget_size_.height() - (double) event->y();
164 
165  x /= widget_size_.width();
166  y /= widget_size_.height();
167 
168  QCPRange key_range = dataset_->KeyRange();
169  QCPRange value_range = dataset_->ValueRange();
170  double key_span = key_range.upper - key_range.lower;
171  double value_span = value_range.upper - value_range.lower;
172 
173  double x_position = x * key_span + key_range.lower;
174  double y_position = y * value_span + value_range.lower;
175 
176 
177  //find nearest x value
178  colvec x_values = dataset_->x();
179  colvec y_values = dataset_->y();
180 
181  double x_diff = x_position - x_values(0);
182  double y_diff = y_position - y_values(0);
183  double x_diff_buf, y_diff_buf, x_value, y_value;
184 
185  for (i = 0; i < x_values.n_rows; ++i){
186  x_diff_buf = x_position - x_values(i);
187  if ((x_diff < 0 && x_diff_buf >= 0) || (x_diff >= 0 && x_diff_buf < 0))
188  break;
189  }
190 
191  if (fabs(x_diff_buf) > fabs(x_position - x_values(i-1)))
192  x_value = x_values(i-1);
193  else
194  x_value = x_values(i);
195 
196  for (i = 0; i < y_values.n_rows; ++i){
197  y_diff_buf = y_position - y_values(i);
198  if ((y_diff < 0 && y_diff_buf >= 0) || (y_diff >= 0 && y_diff_buf < 0))
199  break;
200  }
201 
202  if (fabs(y_diff_buf) > fabs(y_position - y_values(i-1)))
203  y_value = y_values(i-1);
204  else
205  y_value = y_values(i);
206 
207  QVector<int> x_indices;
208  for (i = 0; i < x_values.n_elem; ++i){
209  if (x_values(i) == x_value)
210  x_indices.append(i);
211  }
212  for (i = 0; i < (unsigned int) x_indices.size(); ++i){
213  if (y_values(x_indices[i]) == y_value)
214  break;
215  }
216  current_index_ = x_indices[i];
217 
218  QVector<double> wavelength = dataset_->WavelengthQVector();
219  QVector<double> intensities = dataset_->PointSpectrum(current_index_);
220 
221  current_x_ = x_value;
222  current_y_ = y_value;
223 
224  coordinate_label_->setText("(" +
225  QString::number(current_x_) +
226  ", " +
227  QString::number(current_y_) +
228  ")");
229  value_label_->setText(QString::number(map_data_->results_at_position(current_x_, current_y_)));
230  SetPlot(wavelength, intensities);
231  if (map_data_->univariate_area())
232  SetSecondPlot(map_data_->first_abcissa(), map_data_->first_baseline(current_index_));
233  if (map_data_->univariate_bandwidth()){
234  SetSecondPlot(map_data_->first_abcissa(), map_data_->first_baseline(current_index_));
235  QCPItemLine *mid_line;
236  QVector<double> mid_line_vec;
237  if (spectrum_plot_->itemCount() == 0){
238  QCPItemLine *mid_line = new QCPItemLine(spectrum_plot_);
239  QVector<double> mid_line_vec = map_data_->mid_line(current_index_);
240  mid_line->start->setCoords(mid_line_vec[0], mid_line_vec[1]);
241  mid_line->end->setCoords(mid_line_vec[2], mid_line_vec[3]);
242  mid_line->setPen(QPen(QColor("red")));
243  spectrum_plot_->addItem(mid_line);
244  }
245 
246  }
247  if (map_data_->band_ratio_area()){
248  SetSecondPlot(map_data_->first_abcissa(),
249  map_data_->second_abcissa(),
250  map_data_->first_baseline(current_index_),
251  map_data_->second_baseline(current_index_));
252  }
