56 void CropDialog::on_buttonBox_accepted()
57 {
58 
59  double x_min = x_min_box_->value();
60  double x_max = x_max_box_->value();
61  double y_min = y_min_box_->value();
62  double y_max = y_max_box_->value();
63 
64  if ((x_max - x_min) <= 0 || (y_max - y_min <= 0)){
65  QMessageBox::warning(this, "Error", "Minimum is larger than maximum!");
66  return;
67  }
68 
69  else{
70  dataset_->CropSpectra(x_min, x_max, y_min, y_max);
