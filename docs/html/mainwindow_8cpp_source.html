190 void MainWindow::on_actionNormalize_Standardize_triggered()
191 {
192  int row = dataset_list_widget_->currentRow();
193  QSharedPointer<SpecMap> data = workspace->DatasetAt(row);
194  QStringList methods;
195  methods << tr("Min/Max") << tr("Unit Area") << tr("Z-score");
196  bool ok;
197  QString item = QInputDialog::getItem(this,
198  tr("Normalization/Standardization"),
199  tr("Method:"), methods, 0, false, &ok);
200  if (ok && item == "Min/Max")
201  data->MinMaxNormalize();
202 
203  else if (ok && item == "Unit Area")
204  data->UnitAreaNormalize();
205 
206  else if (ok && item == "Z-score")
207  data->ZScoreNormalize();
208 
209  else
210  return;
