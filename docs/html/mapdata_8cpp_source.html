422 void MapData::DrawScaleBar(double width,
423  double height,
424  QString units,
425  QColor color,
426  QString position,
427  QFont font)
428 {
429  //instantiate shape
430 
431  //QCustomPlot object will take ownership of these and make sure they're deleted.
432  QCPItemRect *scale_bar = new QCPItemRect(map_qcp_);
433  QCPItemText *scale_bar_text = new QCPItemText(map_qcp_);
434 
435 
436  map_qcp_->addItem(scale_bar);
437  map_qcp_->addItem(scale_bar_text);
438 
439  QString text = QString::number(width) + " " + units;
440  scale_bar_text->setText(text);
441  scale_bar_text->setPositionAlignment(Qt::AlignCenter);
442  scale_bar_text->setColor(color);
443  scale_bar_text->setFont(font);
444 
445  QPen pen(color);
446  QBrush brush(color, Qt::SolidPattern);
447  scale_bar->setPen(pen);
448  scale_bar->setBrush(brush);
449  QCPRange key_range = map_->keyAxis()->range();
450  QCPRange value_range = map_->valueAxis()->range();
451 
452  //find corners of image
453  double key_max = key_range.upper;
454  double key_min = key_range.lower;
455  double value_max = value_range.upper;
456  double value_min = value_range.lower;
457 
458  double x_min, y_min, x_max, y_max;
459 
460  if (position == "Top Left"){
461  x_min = key_min + 2.0;
462  x_max = x_min + width;
463  y_max = value_max - 4.0;
464  y_min = y_max - height;
465  }
466 
467  else if (position == "Top Right"){
468  x_max = key_max - 2.0;
469  x_min = x_max - width;
470  y_max = value_max - 4.0;
471  y_min = y_max - height;
472  }
473 
474  else if (position == "Bottom Left"){
475  x_min = key_min + 2.0;
476  x_max = x_min + width;
477  y_min = value_min + 2.0;
478  y_max = y_min + height;
479  }
480 
481  else{
482  x_max = key_max - 2.0;
483  x_min = x_max - width;
484  y_min = value_min + 2.0;
485  y_max = y_min + height;
486  }
487 
488  scale_bar->bottomRight->setCoords(x_max, y_min);
489  scale_bar->topLeft->setCoords(x_min, y_max);
490  double x_mid = ((x_max - x_min) / 2.0) + x_min;
491  double y_text = y_max + 1;
492  if (font.pointSize() >= 10){
493  y_text += (font.pointSize() - 10) / 4;
494  }
495 
496  scale_bar_text->position->setCoords(x_mid, y_text);
497 
498  map_qcp_->show();
499 }

663 void MapData::UseGlobalColorScale(bool arg1)
664 {
665  cout << "MapData::UseGlobalColorScale()" << endl;
666  if (arg1){
667  if (using_global_color_scale_){
668  cout << "arg1 && using_global_color_scale_" << endl;
669  return;
670  }
671  else{
672  cout << "arg1 && !using_global_color_scale_" << endl;
673  QCPColorScale *color_scale = qobject_cast<QCPColorScale *>(map_qcp_->plotLayout()->element(0, 1));
674  color_scale->setGradient(*main_window_->global_gradient());
675  QCPRange data_range = map_->dataRange();
676  main_window_->RecalculateGlobalDataRange(&data_range);
677  color_scale->setDataRange(*main_window_->global_data_range());
678  map_->setGradient(*main_window_->global_gradient());
679  map_qcp_->replot();
680  return;
681  }
682  }
683  else{
684  cout << "!arg1" << endl;
685  if(!using_global_color_scale_){
686  cout << "!arg1 && !using_global_color_scale_" << endl;
687  return;
688  }
689  else{
690  cout << "!arg1 && using_global_color_scale_" << endl;
691  using_global_color_scale_ = false;
692  QCPColorScale *color_scale = qobject_cast<QCPColorScale *>(map_qcp_->plotLayout()->element(0, 1));
693  color_scale->setGradient(gradient_);
694  map_->setGradient(gradient_);
695  color_scale->setDataRange(map_->dataRange());
696  map_qcp_->replot();
697  return;
698  }
699  }
